<script>

  let passActual = "";

  let dataGlobal = [];



  function validarPaso1() {

    const pass = document.getElementById('passInput').value;

    if(!pass) return;

    document.getElementById('loader').style.display = 'block';

    document.getElementById('btnPaso1').disabled = true;



    google.script.run.withSuccessHandler(function(res) {

      document.getElementById('loader').style.display = 'none';

      if(res.error) {

        alert(res.error);

        document.getElementById('btnPaso1').disabled = false;

      } else {

        passActual = pass;

        if(res.indicador === "PERSONAL ADMINISTRATIVO") {

          finalizarConsulta("");

          document.getElementById('paso1').style.display = "none";

        } else {

          document.getElementById('paso1').style.display = "none";

          document.getElementById('seccionCodigo').style.display = "block";

        }

      }

    }).validarPassword(pass);

  }



  function finalizarConsulta(cods) {

    const listaCods = cods !== undefined ? cods : document.getElementById('codigoInput').value;

    if (cods === undefined && !listaCods) { alert("Ingrese al menos un código"); return; }

    

    document.getElementById('loader').style.display = 'block';

    document.getElementById('btnPaso2').disabled = true;

    document.getElementById('resultados').innerHTML = '';



    google.script.run.withSuccessHandler(function(res) {

      document.getElementById('loader').style.display = 'none';

      document.getElementById('btnPaso2').disabled = false;

      if(res.error) { alert(res.error); } 

      else {

        dataGlobal = res.data;

        document.getElementById('contenedorPrincipal').style.maxWidth = "900px";

        document.getElementById('btnExcel').style.display = "inline-block";

        mostrar(res.data);

      }

    }).obtenerDatos(passActual, listaCods);

  }



  function mostrar(data) {

    let html = '';

    const agrupado = data.reduce((acc, obj) => {

      acc[obj.nombre] = acc[obj.nombre] || [];

      acc[obj.nombre].push(obj);

      return acc;

    }, {});



    for (let n in agrupado) {

      const rs = agrupado[n];

      const total = rs.reduce((s, r) => s + r.dias, 0);

      let filas = '';

      rs.forEach(r => {

        filas += `<tr><td class="ps-4 py-2"><b>${r.periodo}</b></td><td class="small">${r.cargo}</td><td class="text-center"><span class="badge bg-danger rounded-pill" style="font-size:0.9rem;">${r.dias}</span></td></tr>`;

      });

      

      html += `

        <div class="employee-card card">

          <div class="card-header-zoom">

            <div>

              <h5 class="mb-1 text-primary fw-bold">${n}</h5>

              <small class="text-muted" style="font-size:0.85rem;">

                Cod: ${rs[0].cod} | Ingreso: ${rs[0].fechaIngreso} | ${rs[0].antiguedad}

              </small>

            </div>

            <div class="total-badge"><span class="d-block" style="font-size:0.7rem;">DÍAS</span><span class="fw-bold">${total}</span></div>

          </div>

          <table class="table table-sm mb-0"><tbody>${filas}</tbody></table>

        </div>`;

    }

    document.getElementById('resultados').innerHTML = html;

  }



  function exportarExcel() {

    if (dataGlobal.length === 0) return;

    const encabezados = ["Subdivisión", "Fecha de Ingreso", "Período", "Antigüedad", "Días", "Cod", "Cargo", "Nombre"];

    const filas = dataGlobal.map(item => [

      item.subdivision, item.fechaIngreso, item.periodo, 

      item.antiguedad.replace("Antigüedad: ", ""), item.dias, item.cod, item.cargo, item.nombre

    ]);

    const ws = XLSX.utils.aoa_to_sheet([encabezados, ...filas]);

    ws['!cols'] = [{wch:25}, {wch:15}, {wch:12}, {wch:15}, {wch:8}, {wch:8}, {wch:25}, {wch:30}];

    const wb = XLSX.utils.book_new();

    XLSX.utils.book_append_sheet(wb, ws, "Data");

    XLSX.writeFile(wb, "Reporte_ZOOM.xlsx");

  }
